public class Activity_Agregar_animales extends AppCompatActivity {

    private Button AgregarDatos;

    private String nomAni = "";
    private String codi = "";
    private String numAre = "";
    private String lot = "";
    private String tipog = "";
    private String sex = "";
    private String raz = "";
    private String colr = "";
    private String propi = "";
    private String carac = "";

    DatabaseReference firebaseDatabase;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity__agregar_animales);

        //Icono en el acción bar
        getSupportActionBar().setDisplayShowHomeEnabled(true);

        Lote=(EditText)findViewById(R.id.et_agre_ani_lote);
        TipoGana=(EditText)findViewById(R.id.sp_agre_ani_tipo);
        Sexo=(EditText)findViewById(R.id.sp_agre_ani_sex);
        Raza=(EditText)findViewById(R.id.sp_agre_ani_raza);
        Color=(EditText)findViewById(R.id.et_agre_ani_color);
        Proposito=(EditText)findViewById(R.id.sp_agre_ani_proposito);
        Caract=(EditText)findViewById(R.id.et_agre_ani_caracte);
        AgregarDatos=(Button)findViewById(R.id.btn_add_datos);

        AgregarDatos.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                nomAni = nombAnimal.getText().toString();
                codi = Codigo.getText().toString();
                numAre = NumeroArete.getText().toString();
                lot = Lote.getText().toString();
                tipog = TipoGana.getText().toString();
                sex = Sexo.getText().toString();
                raz = Raza.getText().toString();
                colr = Color.getText().toString();
                propi = Proposito.getText().toString();
                carac = Caract.getText().toString();


                if (!nomAni.isEmpty()&&!codi.isEmpty()&&!numAre.isEmpty()&&!lot.isEmpty()&&!tipog.isEmpty()&&!sex.isEmpty()&&!raz.isEmpty()&&!colr.isEmpty()&&!propi.isEmpty()&&!carac.isEmpty()){

                    if (codi.length() >=5){

                        Map<String,Object> map = new HashMap<>();
                        map.put("nombreanimal", nomAni);
                        map.put("codigo", codi);
                        map.put("numerodearete", numAre);
                        map.put("lote", lot);
                        map.put("tipodeganado", tipog);
                        map.put("sexo", sex);
                        map.put("raza", raz);
                        map.put("color", colr);
                        map.put("proposito", propi);

                        firebaseDatabase.child("Animales").push().setValue(map);

                        Toast.makeText(Activity_Agregar_animales.this,"Datos agregados correctamente ",Toast.LENGTH_SHORT).show();

package com.example.agro;

import androidx.annotation.NonNull;
import androidx.appcompat.app.AppCompatActivity;

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.Toast;

import com.google.android.gms.tasks.OnCompleteListener;
import com.google.android.gms.tasks.Task;
import com.google.firebase.auth.AuthResult;
import com.google.firebase.auth.FirebaseAuth;

public class Activity_cuatro extends AppCompatActivity {

    private EditText mail;
    private EditText passw;
    private Button init;
    private Button resetpass;

    private String cor = "";
    private String pas = "";

    private FirebaseAuth mAuth;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_cuatro);

        //Icono en el acción bar
        getSupportActionBar().setDisplayShowHomeEnabled(true);
        getSupportActionBar().setIcon(R.mipmap.ic_launcher);

        mAuth = FirebaseAuth.getInstance();

        mail =(EditText)findViewById(R.id.et_mail_prin);
        passw =(EditText)findViewById(R.id.et_pass_prin);
        init =(Button) findViewById(R.id.btn_login);
        resetpass =(Button) findViewById(R.id.btn_recuperar);

        init.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                cor= mail.getText().toString();
                pas= passw.getText().toString();

                if (!cor.isEmpty()&& !pas.isEmpty()){
                    loginUser();
                }else{
                    Toast.makeText(Activity_cuatro.this,"Complete los campos",Toast.LENGTH_SHORT).show();
                }
            }
        });

        resetpass.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                startActivity(new Intent(Activity_cuatro.this, Activity_quinta.class));
            }
        });

    }

    private void loginUser() {
        mAuth.signInWithEmailAndPassword(cor, pas).addOnCompleteListener(new OnCompleteListener<AuthResult>() {
            @Override
            public void onComplete(@NonNull Task<AuthResult> task) {
                if (task.isSuccessful()) {
                    startActivity(new Intent(Activity_cuatro.this, Activity_seis.class));
                    finish();
                }else{
                    Toast.makeText(Activity_cuatro.this, "Usuario y/o Contraseña Incorrectas", Toast.LENGTH_SHORT).show();
                }
            }
        });
    }

    @Override
    protected void onStart() {
        super.onStart();

        if (mAuth.getCurrentUser() !=null){
            startActivity(new Intent(Activity_cuatro.this, Activity_seis.class));
            finish();
        }

    }
    //Metodo para el boton volver(icono)
    public void Volver(View view){
        Intent volver = new Intent(this, Activity_sig.class);
        startActivity(volver);
    }



}

import androidx.annotation.NonNull;

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;

import com.google.android.gms.tasks.Task;
import com.google.firebase.auth.FirebaseAuth;

public class Activity_quinta extends AppCompatActivity {

    private EditText mail;
    private Button recupe;

    private String email = "";

    private FirebaseAuth mAuth;

    private ProgressDialog mDialog;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_quinta);

        //Icono en el acción bar
        getSupportActionBar().setDisplayShowHomeEnabled(true);
        getSupportActionBar().setIcon(R.mipmap.ic_launcher);

        mAuth = FirebaseAuth.getInstance();
        mDialog = new ProgressDialog(this);

        mail = (EditText)findViewById(R.id.et_mail_recu);
        recupe = (Button) findViewById(R.id.btn_enviar_recu);

        recupe.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {

                email = mail.getText().toString();

                if (!email.isEmpty()){
                    mDialog.setMessage("Espere un momento.......");
                    mDialog.setCanceledOnTouchOutside(false);
                    mDialog.show();
                    resetPassword();

                }else{
                    Toast.makeText(Activity_quinta.this, "Debe ingresar el correo", Toast.LENGTH_SHORT).show();
                }

                mDialog.dismiss();

            }
        });


    }


    private void resetPassword(){

        mAuth.setLanguageCode("es");
        mAuth.sendPasswordResetEmail(email).addOnCompleteListener(new OnCompleteListener<Void>() {
            @Override
            public void onComplete(@NonNull Task<Void> task) {
                if (task.isSuccessful()){
                    Toast.makeText(Activity_quinta.this, "Se ha enviando un correo para restablecer la contraseña", Toast.LENGTH_SHORT).show();
                }else{
                    Toast.makeText(Activity_quinta.this, "No se pudo enviar el correo", Toast.LENGTH_SHORT).show();
                }
            }
        });

    }

    //Metodo para el boton volver(icono)
    public void Volver(View view){
        Intent volver = new Intent(this, Activity_sig.class);
        startActivity(volver);
    }

            @Override
            public void onClick(View view) {

                startActivity(new Intent(Activity_seis.this, Activity_cuatro.class));
                finish();
            }
        });

        getUserinfo();
    }

    private void getUserinfo() {
        String id = mAuth.getCurrentUser().getUid();
        mDatabase.child("Users").child(id).addValueEventListener(new ValueEventListener() {
            @Override
            public void onDataChange(@NonNull DataSnapshot snapshot) {
                if (snapshot.exists()){
                    String nombre = snapshot.child("nombre").getValue().toString();

                    nombr.setText(nombre);
                }
            }

            @Override
            public void onCancelled(@NonNull DatabaseError error) {

            }
        });
    }

package com.example.agro;

import androidx.annotation.NonNull;
import androidx.appcompat.app.AppCompatActivity;

import android.content.Intent;
import android.icu.text.UnicodeSetSpanner;
import android.media.MediaPlayer;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.Toast;

import com.google.android.gms.tasks.OnCompleteListener;
import com.google.android.gms.tasks.Task;
import com.google.firebase.auth.AuthResult;
import com.google.firebase.auth.FirebaseAuth;
import com.google.firebase.database.DatabaseReference;
import com.google.firebase.database.FirebaseDatabase;

import java.util.HashMap;
import java.util.Map;

public class Activity_tres extends AppCompatActivity  {

    private EditText nomb, apell, cedl, corre, passw;
    private Button Registros;

    //Variables de los datos que vamos a registrar

    private String nom = "";
    private String ape = "";
    private String ced = "";
    private String cor = "";
    private String pas = "";

    FirebaseAuth mAuth;
    DatabaseReference mDatabase;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_tres);

        //Icono en el acción bar
        getSupportActionBar().setDisplayShowHomeEnabled(true);
        getSupportActionBar().setIcon(R.mipmap.ic_launcher);

        mAuth = FirebaseAuth.getInstance();
        mDatabase = FirebaseDatabase.getInstance().getReference();

        nomb=(EditText)findViewById(R.id.et_name);
        apell=(EditText)findViewById(R.id.et_last);
        cedl=(EditText)findViewById(R.id.en_cc);
        corre=(EditText)findViewById(R.id.et_email);
        passw=(EditText)findViewById(R.id.et_pass);
        Registros=(Button)findViewById(R.id.btn_regis_1);

        Registros.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                nom = nomb.getText().toString();
                ape = apell.getText().toString();
                ced = cedl.getText().toString();
                cor = corre.getText().toString();
                pas = passw.getText().toString();


                if (!nom.isEmpty()&&!ape.isEmpty()&&!ced.isEmpty()&&!cor.isEmpty()&&!pas.isEmpty()){

                    if (pas.length() >=6){
                        registrarUser();
                    }else{
                        Toast.makeText(Activity_tres.this,"La contraseña sebe tener al menos 6 caracteres",Toast.LENGTH_SHORT).show();
                    }


                }else{
                    Toast.makeText(Activity_tres.this,"Debe Completar todos los Campos",Toast.LENGTH_SHORT).show();
                }
            }
        });
    }
    //Metodo para el boton volver(icono)
    public void Volver(View view){
        Intent volver = new Intent(this, Activity_sig.class);
        startActivity(volver);
    }

    private void registrarUser(){
        mAuth.createUserWithEmailAndPassword(cor, pas).addOnCompleteListener(new OnCompleteListener<AuthResult>() {
            @Override
            public void onComplete(@NonNull Task<AuthResult> task) {
            if (task.isSuccessful()){

                Map<String,Object> map = new HashMap<>();
                map.put("nombre", nom);
                map.put("apellido", ape);
                map.put("cedula", ced);
                map.put("correo", cor);
                map.put("clave", pas);

                String id = mAuth.getCurrentUser().getUid();

                mDatabase.child("Users").child(id).setValue(map).addOnCompleteListener(new OnCompleteListener<Void>() {
                    @Override
                    public void onComplete(@NonNull Task<Void> task2) {
                        if (task2.isSuccessful()){
                            startActivity(new Intent(Activity_tres.this, Activity_seis.class));
                            finish();
                        }else{
                            Toast.makeText(Activity_tres.this,"No se puedieron crear los datos de manera correcta",Toast.LENGTH_SHORT).show();
                        }
                    }
                });

            }else{
                Toast.makeText(Activity_tres.this,"No se pudo registrar este usuario",Toast.LENGTH_SHORT).show();
            }
            }
        });
    }
